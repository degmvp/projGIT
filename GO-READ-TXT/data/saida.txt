// Go program to illustrate the 
// use of nested if statement
package main
import "fmt"
  
func main() {
      
   // taking two local variable
   var v1 int = 400
   var v2 int = 700
   
   // using if statement
   if( v1 == 400 ) {
         
      // if condition is true then 
      // check the following 
      if( v2 == 700 )  {
            
         // if condition is true 
         // then display the following 
         fmt.Printf("Value of v1 is 400 and v2 is 700\n" );
      }
   }
    
}
